SCENARIO 1: Feed & Posts

Query 1
Show all posts made by verified users.

SELECT u.username, p.post_id, p.caption, p.location_name
FROM Posts p
JOIN Users u ON p.user_id = u.user_id
WHERE u.is_verified = TRUE;


Query 2
Show all posts that have more than one media file (i.e., posts with multiple images or videos).

SELECT p.post_id, p.caption, u.username,
       COUNT(pm.media_id) AS media_count
FROM Posts p
JOIN Post_Media pm ON p.post_id = pm.post_id
JOIN Users u ON p.user_id = u.user_id
GROUP BY p.post_id, p.caption, u.username
HAVING COUNT(pm.media_id) > 1
ORDER BY media_count DESC;


Query 3
Show all posts that are from Mumbai location.

SELECT u.username, p.post_id, p.caption,
       p.location_name
FROM Posts p
JOIN Users u ON p.user_id = u.user_id
WHERE p.location_name ILIKE '%Mumbai%';


Query 4
Show the total number of posts made by each user.

SELECT u.username, COUNT(p.post_id) AS total_posts
FROM Users u
JOIN Posts p ON u.user_id = p.user_id
GROUP BY u.user_id, u.username
ORDER BY total_posts DESC;


Query 5
Show posts along with the number of media files attached to each post.

SELECT p.post_id, u.username, p.caption, COUNT(pm.media_id) AS media_count
FROM Posts p
JOIN Users u ON p.user_id = u.user_id
LEFT JOIN Post_Media pm ON p.post_id = pm.post_id
GROUP BY p.post_id, u.username, p.caption
ORDER BY media_count DESC;



SCENARIO 2: Likes & Comments (Engagement)

Query 1
Show the top 10 most liked posts along with the username of the person who posted them.

SELECT p.post_id, p.caption, u.username,
       COUNT(l.like_id) AS total_likes
FROM Posts p
JOIN Likes l ON p.post_id = l.target_id
JOIN target_type tt ON l.target_type_id = tt.target_type_id
JOIN Users u ON p.user_id = u.user_id
WHERE tt.target_type_name = 'post'
GROUP BY p.post_id, p.caption, u.username
ORDER BY total_likes DESC
LIMIT 10;


Query 2
Display all comments on a post, including main comments and their direct replies together.

SELECT
    c_main.comment_id,
    u1.username AS commenter,
    c_main.content AS comment,
    c_reply.comment_id AS reply_id,
    u2.username AS replier,
    c_reply.content AS reply
FROM Comments c_main
LEFT JOIN Comments c_reply
    ON c_reply.parent_comment_id = c_main.comment_id
LEFT JOIN Users u1 ON c_main.user_id = u1.user_id
LEFT JOIN Users u2 ON c_reply.user_id = u2.user_id
WHERE c_main.post_id = 1
  AND c_main.parent_comment_id IS NULL
ORDER BY c_main.comment_id;


Query 3
Find users who have liked posts but have not commented on them.

SELECT DISTINCT u.username
FROM Likes l
JOIN target_type tt ON l.target_type_id = tt.target_type_id
JOIN Users u ON l.user_id = u.user_id
WHERE tt.target_type_name = 'post'
  AND l.target_id = 1
  AND NOT EXISTS (
    SELECT 1 FROM Comments c
    WHERE c.post_id = l.target_id
      AND c.user_id = l.user_id
  );


Query 4
Count the total number of comments each user has received on all their posts.

SELECT u.username, COUNT(c.comment_id) AS total_comments_received
FROM Users u
JOIN Posts p ON u.user_id = p.user_id
JOIN Comments c ON p.post_id = c.post_id
GROUP BY u.user_id, u.username
ORDER BY total_comments_received DESC;


Query 5
Find posts where users have liked their own posts, representing self-engagement activity.

SELECT p.post_id, p.caption, u.username
FROM Posts p
JOIN Users u ON p.user_id = u.user_id
JOIN Likes l ON l.target_id = p.post_id
JOIN target_type tt ON l.target_type_id = tt.target_type_id
WHERE tt.target_type_name = 'post'
  AND l.user_id = p.user_id;



SCENARIO 3: Stories & Story Views

Query 1
Show all stories that are still active (not expired), along with the user who posted them and their audience type.

SELECT u.username, s.story_id, s.media_url,
       s.expires_at, at.audience_type_name
FROM Stories s
JOIN Users u ON s.user_id = u.user_id
JOIN Audience_type at ON s.audience_type_id = at.audience_type_id
WHERE s.expires_at > NOW()
ORDER BY s.expires_at ASC;


Query 2
Count how many unique users have viewed each active story.

SELECT s.story_id, u.username AS story_owner,
       COUNT(sv.viewer_id) AS view_count
FROM Stories s
JOIN Story_views sv ON s.story_id = sv.story_id
JOIN Users u ON s.user_id = u.user_id
WHERE s.expires_at > NOW()
GROUP BY s.story_id, u.username
ORDER BY view_count DESC;


Query 3
Show stories that were viewed only by users who follow the story owner, indicating genuine follower engagement.

SELECT s.story_id, u_owner.username AS owner,
       u_viewer.username AS viewer
FROM Story_views sv
JOIN Stories s ON sv.story_id = s.story_id
JOIN Users u_owner ON s.user_id = u_owner.user_id
JOIN Users u_viewer ON sv.viewer_id = u_viewer.user_id
WHERE EXISTS (
    SELECT 1 FROM Followers f
    WHERE f.follower_id = sv.viewer_id
      AND f.following_id = s.user_id
)
ORDER BY s.story_id;


Query 4
Find the top 5 users who view the most stories.

SELECT u.username, COUNT(sv.story_id) AS stories_watched
FROM Story_views sv
JOIN Users u ON sv.viewer_id = u.user_id
GROUP BY u.user_id, u.username
ORDER BY stories_watched DESC
LIMIT 5;


Query 5
Find users who posted a story but did not receive any views.

SELECT u.username, s.story_id, s.expires_at
FROM Stories s
JOIN Users u ON s.user_id = u.user_id
WHERE NOT EXISTS (
    SELECT 1 FROM Story_views sv
    WHERE sv.story_id = s.story_id
)
ORDER BY s.expires_at DESC;



SCENARIO 4: Followers & Social Connections

Query 1
Find pairs of users who follow each other (mutual followers).

SELECT u1.username AS user1, u2.username AS user2
FROM Followers f1
JOIN Followers f2
    ON f1.follower_id = f2.following_id
   AND f1.following_id = f2.follower_id
JOIN Users u1 ON f1.follower_id = u1.user_id
JOIN Users u2 ON f1.following_id = u2.user_id
WHERE f1.follower_id < f1.following_id;


Query 2
Show the number of followers for each verified user.

SELECT u.username, u.is_verified,
       COUNT(f.follower_id) AS follower_count
FROM Users u
LEFT JOIN Followers f ON u.user_id = f.following_id
WHERE u.is_verified = TRUE
GROUP BY u.user_id, u.username, u.is_verified
ORDER BY follower_count DESC;


Query 3
Find users who follow someone who has blocked them (an edge case in the system).

SELECT u_follower.username AS follower,
       u_followed.username AS followed_and_blocked
FROM Followers f
JOIN Users u_follower ON f.follower_id = u_follower.user_id
JOIN Users u_followed ON f.following_id = u_followed.user_id
WHERE EXISTS (
    SELECT 1 FROM Blocked_Accounts ba
    WHERE ba.blocker_id = f.following_id
      AND ba.blocked_id = f.follower_id
);


Query 4
List users who have more than 4 followers along with their account type.

SELECT u.username, act.type_name AS account_type,
       COUNT(f.follower_id) AS followers
FROM Users u
JOIN Followers f ON u.user_id = f.following_id
JOIN Account_Types act ON u.account_type_id = act.type_id
GROUP BY u.user_id, u.username, act.type_name
HAVING COUNT(f.follower_id) > 4
ORDER BY followers DESC;


Query 5
Suggest users that user 1 may know based on mutual connections (followers of people they follow).

SELECT DISTINCT u.username, u.user_id
FROM Followers f1
JOIN Followers f2 ON f1.following_id = f2.follower_id
JOIN Users u ON f2.following_id = u.user_id
WHERE f1.follower_id = 1
  AND f2.following_id <> 1
  AND f2.following_id NOT IN (
      SELECT following_id FROM Followers WHERE follower_id = 1
  )
LIMIT 10;



SCENARIO 5: Chats & Calls

Query 1
Show the most recent message in each conversation between users (like an inbox preview).

SELECT DISTINCT ON (LEAST(ch.sender_id, ch.receiver_id),
                    GREATEST(ch.sender_id, ch.receiver_id))
    u1.username AS sender, u2.username AS receiver,
    ch.message_content, ch.send_at
FROM Chats ch
JOIN Users u1 ON ch.sender_id = u1.user_id
JOIN Users u2 ON ch.receiver_id = u2.user_id
ORDER BY LEAST(ch.sender_id, ch.receiver_id),
         GREATEST(ch.sender_id, ch.receiver_id),
         ch.send_at DESC;


Query 2
Count the number of unread messages for each user.

SELECT u.username AS receiver,
       COUNT(ch.chat_id) AS unread_messages
FROM Chats ch
JOIN Users u ON ch.receiver_id = u.user_id
WHERE ch.is_read = FALSE
GROUP BY u.user_id, u.username
ORDER BY unread_messages DESC;


Query 3
Calculate the average call duration for each call type (audio or video).

SELECT ct.type_name AS call_type,
       ROUND(AVG(c.duration_seconds)) AS avg_seconds,
       COUNT(c.call_id) AS total_calls
FROM Calls c
JOIN Call_Types ct ON c.call_type_id = ct.call_type_id
WHERE c.status = 'completed'
GROUP BY ct.type_name
ORDER BY avg_seconds DESC;


Query 4
Find pairs of users who have exchanged messages both ways (two-way conversation).

SELECT u1.username AS person1, u2.username AS person2
FROM Chats ch1
JOIN Chats ch2
    ON ch1.sender_id = ch2.receiver_id
   AND ch1.receiver_id = ch2.sender_id
JOIN Users u1 ON ch1.sender_id = u1.user_id
JOIN Users u2 ON ch1.receiver_id = u2.user_id
WHERE ch1.sender_id < ch1.receiver_id
GROUP BY u1.username, u2.username;


Query 5
Find users who had missed calls (calls that were not answered).

SELECT u_caller.username AS caller,
       u_receiver.username AS receiver,
       c.started_at
FROM Calls c
JOIN Users u_caller ON c.caller_id = u_caller.user_id
JOIN Users u_receiver ON c.receiver_id = u_receiver.user_id
WHERE c.status = 'missed'
ORDER BY c.started_at DESC;



SCENARIO 6: Reels & Music

Query 1
Show the top 5 most viewed reels along with their music details.

SELECT r.reel_id, u.username AS creator,
       ml.song_title, ml.artist_name,
       r.total_views
FROM Reels r
JOIN Users u ON r.user_id = u.user_id
JOIN Music_Library ml ON r.music_id = ml.music_id
ORDER BY r.total_views DESC
LIMIT 5;


Query 2
Find the top 3 songs that are used most frequently in reels.

SELECT ml.song_title, ml.artist_name,
       COUNT(r.reel_id) AS used_in_reels
FROM Music_Library ml
JOIN Reels r ON ml.music_id = r.music_id
GROUP BY ml.music_id, ml.song_title, ml.artist_name
ORDER BY used_in_reels DESC
LIMIT 3;


Query 3
List all reels that allow remixing (i.e., marked as remixable).

SELECT r.reel_id, u.username, ml.song_title,
       r.total_views, r.is_remixable
FROM Reels r
JOIN Users u ON r.user_id = u.user_id
JOIN Music_Library ml ON r.music_id = ml.music_id
WHERE r.is_remixable = TRUE
ORDER BY r.total_views DESC;


Query 4
Find songs that are not used in any reel yet.

SELECT ml.music_id, ml.song_title, ml.artist_name
FROM Music_Library ml
WHERE NOT EXISTS (
    SELECT 1 FROM Reels r
    WHERE r.music_id = ml.music_id
)
ORDER BY ml.song_title;


Query 5
Show reels that have more than 5 views and are also remixable.

SELECT r.reel_id, u.username, ml.song_title,
       r.total_views
FROM Reels r
JOIN Users u ON r.user_id = u.user_id
JOIN Music_Library ml ON r.music_id = ml.music_id
WHERE r.total_views > 5
  AND r.is_remixable = TRUE
ORDER BY r.total_views DESC;



SCENARIO 7: User Analytics & Insights

Query 1
Find the user who has the highest total activity based on posts, reels, and stories combined.

SELECT u.username,
       (SELECT COUNT(*) FROM Posts WHERE user_id = u.user_id) AS post_count,
       (SELECT COUNT(*) FROM Reels WHERE user_id = u.user_id) AS reel_count,
       (SELECT COUNT(*) FROM Stories WHERE user_id = u.user_id) AS story_count,
       (SELECT COUNT(*) FROM Posts WHERE user_id = u.user_id) +
       (SELECT COUNT(*) FROM Reels WHERE user_id = u.user_id) +
       (SELECT COUNT(*) FROM Stories WHERE user_id = u.user_id) AS total_activity
FROM Users u
ORDER BY total_activity DESC
LIMIT 1;


Query 2
Find users who have added each other to their close friends list (mutual close friends).

SELECT u1.username AS person1, u2.username AS person2
FROM Close_Friends_List cfl1
JOIN Close_Friends_List cfl2
    ON cfl1.user_id = cfl2.friend_id
   AND cfl1.friend_id = cfl2.user_id
JOIN Users u1 ON cfl1.user_id = u1.user_id
JOIN Users u2 ON cfl1.friend_id = u2.user_id
WHERE cfl1.user_id < cfl1.friend_id;


Query 3
Show the number of highlights created by each verified user.

SELECT u.username,
       COUNT(h.highlight_id) AS highlight_count
FROM Users u
LEFT JOIN Highlights h ON u.user_id = h.user_id
WHERE u.is_verified = TRUE
GROUP BY u.user_id, u.username
ORDER BY highlight_count DESC;


Query 4
Find users who have blocked someone but are still being followed by that same person.

SELECT u_blocker.username AS blocker,
       u_blocked.username AS blocked_still_following
FROM Blocked_Accounts ba
JOIN Users u_blocker ON ba.blocker_id = u_blocker.user_id
JOIN Users u_blocked ON ba.blocked_id = u_blocked.user_id
WHERE EXISTS (
    SELECT 1 FROM Followers f
    WHERE f.follower_id = ba.blocked_id
      AND f.following_id = ba.blocker_id
);


Query 5
Rank users based on total engagement received on their posts (likes + comments).

SELECT u.username,
       COUNT(DISTINCT l.like_id) AS likes_received,
       COUNT(DISTINCT c.comment_id) AS comments_received,
       COUNT(DISTINCT l.like_id) + COUNT(DISTINCT c.comment_id) AS engagement_score
FROM Users u
LEFT JOIN Posts p ON u.user_id = p.user_id
LEFT JOIN Likes l ON l.target_id = p.post_id
LEFT JOIN Comments c ON c.post_id = p.post_id
GROUP BY u.user_id, u.username
ORDER BY engagement_score DESC;



COMBINED QUERIES

Query 1
Find users who have uploaded stories but have not saved any of them into highlights.

SELECT DISTINCT
    u.user_id,
    u.username
FROM Users u
JOIN Stories s ON u.user_id = s.user_id
WHERE NOT EXISTS (
    SELECT 1
    FROM Highlights h
    WHERE h.user_id = u.user_id
)
ORDER BY u.username;


Query 2
Find users who have created at least one post, one story, and one reel.

SELECT
    u.user_id,
    u.username
FROM Users u
WHERE EXISTS (
    SELECT 1 FROM Posts p WHERE p.user_id = u.user_id
)
AND EXISTS (
    SELECT 1 FROM Stories s WHERE s.user_id = u.user_id
)
AND EXISTS (
    SELECT 1 FROM Reels r WHERE r.user_id = u.user_id
)
ORDER BY u.username;
